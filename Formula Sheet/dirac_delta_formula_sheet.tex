\QuadBox{(TL)}{QCBlue}{1) Defining Properties}{
\[
\int_{-\infty}^{\infty} f(x)\delta(x-a)\,dx=f(a)
\]
\[
\int_{-\infty}^{\infty}\delta(x-a)\,dx=1
\]
\[
\delta(ax)=\frac{1}{|a|}\delta(x)
\]
\[
\delta(g(x))=\sum_i \frac{\delta(x-x_i)}{|g'(x_i)|}
\]
for simple zeros $x_i$ of $g(x)$.
}
\QuadBox{(TR)}{QCGreen}{2) Higher Dimensions}{
\[
\delta^{(2)}(\mathbf r-\mathbf r_0)=\delta(x-x_0)\delta(y-y_0)
\]
\[
\delta^{(3)}(\mathbf r-\mathbf r_0)=\delta(x-x_0)\delta(y-y_0)\delta(z-z_0)
\]
Spherical coordinates:
\[
\delta^{(3)}(\mathbf r-\mathbf r_0)=\frac{1}{r^2\sin\theta}\delta(r-r_0)\delta(\theta-\theta_0)\delta(\phi-\phi_0)
\]
Cylindrical coordinates:
\[
\delta^{(3)}(\mathbf r-\mathbf r_0)=\frac{1}{s}\delta(s-s_0)\delta(\phi-\phi_0)\delta(z-z_0)
\]
}
\QuadBox{(BL)}{QCOrange}{3) Electrodynamics Uses}{
Point charge:
\[
\rho(\mathbf r)=q\delta^{(3)}(\mathbf r-\mathbf r_0)
\]
Line charge on $z$ axis:
\[
\rho(x,y,z)=\lambda \delta(x)\delta(y)
\]
Plane charge at $z=0$:
\[
\rho(x,y,z)=\sigma\delta(z)
\]
Derivative identity:
\[
\int f(x)\delta'(x-a)\,dx=-f'(a)
\]
}
\QuadBox{(BR)}{QCPurple}{4) Delta Sequences}{
Gaussian approximation:
\[
\delta_\varepsilon(x)=\frac{1}{\sqrt{\pi}\varepsilon}e^{-x^2/\varepsilon^2}
\]
\[
\int_{-\infty}^{\infty}\delta_\varepsilon(x)\,dx=1
\]
Distributional limit:
\[
\lim_{\varepsilon\to 0}\int f(x)\delta_\varepsilon(x-a)\,dx=f(a)
\]
Reminder:

The delta object is a distribution, not an ordinary pointwise function.
}
